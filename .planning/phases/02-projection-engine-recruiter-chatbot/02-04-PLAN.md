<!-- Chat API contract from Plan 02-03 -->

POST /chat
Request:
```json
{
  "message": "string — the recruiter's question",
  "history": [
    { "role": "user", "content": "previous message" },
    { "role": "assistant", "content": "previous response" }
  ]
}
```
Headers:
  Authorization: Bearer {token}
  Content-Type: application/json

Response: Streaming text/plain via ReadableStream (Hono streamText)
- Chunks arrive as raw text (not SSE format)
- Read via response.body.getReader() + TextDecoder

GET /health
Response: { "status": "ok", "service": "breakfast-club-chatbot" }

Error responses: { "error": "string" } with 400/401/403/500 status

    3. **sendMessage function:**
       ```javascript
       async function sendMessage(message) {
         // Disable input + button
         const input = document.getElementById("message-input");
         const btn = document.getElementById("send-btn");
         input.disabled = true;
         btn.disabled = true;
         document.getElementById("status").textContent = "Thinking...";

         // Add user message to UI
         appendMessage("user", message);
         history.push({ role: "user", content: message });

         // Create assistant message bubble (will be streamed into)
         const assistantEl = appendMessage("assistant", "");

         try {
           const response = await fetch("/chat", {
             method: "POST",
             headers: {
               "Content-Type": "application/json",
               "Authorization": token ? `Bearer ${token}` : "",
             },
             body: JSON.stringify({ message, history: history.slice(0, -1) }),
           });

           if (!response.ok) {
             const err = await response.json().catch(() => ({ error: "Request failed" }));
             assistantEl.textContent = `Error: ${err.error || response.statusText}`;
             return;
           }

           // Stream the response
           const reader = response.body.getReader();
           const decoder = new TextDecoder();
           let fullResponse = "";

           while (true) {
             const { done, value } = await reader.read();
             if (done) break;
             const chunk = decoder.decode(value, { stream: true });
             fullResponse += chunk;
             assistantEl.textContent = fullResponse;
             // Auto-scroll to bottom
             document.getElementById("messages").scrollTop =
               document.getElementById("messages").scrollHeight;
           }

           history.push({ role: "assistant", content: fullResponse });
         } catch (error) {
           assistantEl.textContent = `Error: ${error.message}`;
         } finally {
           input.disabled = false;
           btn.disabled = false;
           input.focus();
           document.getElementById("status").textContent = "Ready";
         }
       }
       ```

    4. **appendMessage helper:**
       ```javascript
       function appendMessage(role, content) {
         const messages = document.getElementById("messages");
         const div = document.createElement("div");
         div.className = `message ${role}`;
         div.textContent = content;
         messages.appendChild(div);
         messages.scrollTop = messages.scrollHeight;
         return div;
       }
       ```

    5. **Form handler:**
       ```javascript
       document.getElementById("chat-form").addEventListener("submit", (e) => {
         e.preventDefault();
         const input = document.getElementById("message-input");
         const message = input.value.trim();
         if (!message) return;
         input.value = "";
         sendMessage(message);
       });
       ```

       });
